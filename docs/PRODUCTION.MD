Architecture Patterns                                                                                                                                                                                     
                                                                                                                                                                                                            
  Clean Architecture                                                                                                                                                                                        
                                                                                                                                                                                                            
  Structure applications with clear separation of concerns:                                                                                                                                                 
                                                                                                                                                                                                            
  Layers                                                                                                                                                                                                    
                                                                                                                                                                                                            
  - Entities: Business objects with core business rules                                                                                                                                                     
  - Use Cases: Application-specific business rules                                                                                                                                                          
  - Interface Adapters: Convert data between layers                                                                                                                                                         
  - Frameworks & Drivers: External frameworks and tools                                                                                                                                                     
                                                                                                                                                                                                            
  Benefits                                                                                                                                                                                                  
                                                                                                                                                                                                            
  - Independent of frameworks                                                                                                                                                                               
  - Testable without external dependencies                                                                                                                                                                  
  - Independent of UI or database                                                                                                                                                                           
  - Maintainable and flexible                                                                                                                                                                               
                                                                                                                                                                                                            
  Implementation                                                                                                                                                                                            
                                                                                                                                                                                                            
  Place use cases in separate modules:                                                                                                                                                                      
                                                                                                                                                                                                            
  // useCases/CreateUser.js                                                                                                                                                                                 
  class CreateUser {                                                                                                                                                                                        
    constructor(userRepository, emailService) {                                                                                                                                                             
      this.userRepository = userRepository;                                                                                                                                                                 
      this.emailService = emailService;                                                                                                                                                                     
    }                                                                                                                                                                                                       
                                                                                                                                                                                                            
    async execute(userData) {                                                                                                                                                                               
      const user = new User(userData);                                                                                                                                                                      
      await this.userRepository.save(user);                                                                                                                                                                 
      await this.emailService.sendWelcome(user.email);                                                                                                                                                      
      return user;                                                                                                                                                                                          
    }                                                                                                                                                                                                       
  }

  Hexagonal Architecture (Ports and Adapters)                                                                                                                                                               
                                                                                                                                                                                                            
  Structure around business logic with interchangeable adapters:                                                                                                                                            
                                                                                                                                                                                                            
  Components                                                                                                                                                                                                
                                                                                                                                                                                                            
  - Core: Business logic and entities                                                                                                                                                                       
  - Ports: Interfaces defining required or provided services                                                                                                                                                
  - Adapters: Concrete implementations of ports                                                                                                                                                             
                                                                                                                                                                                                            
  Example Structure                                                                                                                                                                                         
                                                                                                                                                                                                            
  src/                                                                                                                                                                                                      
  ├── core/                                                                                                                                                                                                 
  │   ├── domain/                                                                                                                                                                                           
  │   │   ├── entities/                                                                                                                                                                                     
  │   │   └── services/                                                                                                                                                                                     
  │   └── usecases/                                                                                                                                                                                         
  ├── adapters/                                                                                                                                                                                             
  │   ├── primary/                                                                                                                                                                                          
  │   │   └── rest/                                                                                                                                                                                         
  │   └── secondary/                                                                                                                                                                                        
  │       ├── database/                                                                                                                                                                                     
  │       └── external/                                                                                                                                                                                     
  └── ports/                                                                                                                                                                                                
      ├── inbound/                                                                                                                                                                                          
      └── outbound/

  Domain-Driven Design (DDD)                                                                                                                                                                                
                                                                                                                                                                                                            
  Model software around business domains:                                                                                                                                                                   
                                                                                                                                                                                                            
  Building Blocks                                                                                                                                                                                           
                                                                                                                                                                                                            
  - Aggregate: Cluster of domain objects treated as a unit                                                                                                                                                  
  - Entity: Objects with distinct identity                                                                                                                                                                  
  - Value Object: Objects defined by attributes, not identity                                                                                                                                               
  - Domain Event: Record of something that happened                                                                                                                                                         
  - Repository: Collection-like interface for aggregates                                                                                                                                                    
  - Service: Operations that don't naturally belong to entities                                                                                                                                             
                                                                                                                                                                                                            
  Example                                                                                                                                                                                                   
                                                                                                                                                                                                            
  // domain/models/Order.js (Aggregate Root)                                                                                                                                                                
  class Order {                                                                                                                                                                                             
    constructor(customerId) {                                                                                                                                                                               
      this.id = uuid.v4();                                                                                                                                                                                  
      this.customerId = customerId;                                                                                                                                                                         
      this.items = [];                                                                                                                                                                                      
      this.status = 'pending';                                                                                                                                                                              
      this.version = 0;                                                                                                                                                                                     
    }                                                                                                                                                                                                       
                                                                                                                                                                                                            
    addItem(product, quantity) {                                                                                                                                                                            
      if (this.status !== 'pending') {                                                                                                                                                                      
        throw new Error('Cannot add items to non-pending order');                                                                                                                                           
      }                                                                                                                                                                                                     
      this.items.push(new OrderItem(product, quantity));                                                                                                                                                    
      this.version++;                                                                                                                                                                                       
      return new OrderItemAddedEvent(this.id, product, quantity);                                                                                                                                           
    }                                                                                                                                                                                                       
  }                                                                                                                                                                                                         
                                                                                                                                                                                                            
  Layering                                                                                                                                                                                                  
                                                                                                                                                                                                            
  - Infrastructure: Database, messaging, HTTP frameworks                                                                                                                                                    
  - Domain: Business logic, entities, domain services                                                                                                                                                       
  - Application: Use cases, transactions, security                                                                                                                                                          
  - Presentation: API controllers, UI                                                                                                                                                                       
                                                                                                                                                                                                            
  Microservices Patterns                                                                                                                                                                                    
                                                                                                                                                                                                            
  Design distributed systems with clear boundaries:                                                                                                                                                         
                                                                                                                                                                                                            
  Service Decomposition                                                                                                                                                                                     
                                                                                                                                                                                                            
  - Split by business capability or domain                                                                                                                                                                  
  - Each service owns its data and logic                                                                                                                                                                    
  - Loose coupling and high cohesion                                                                                                                                                                        
  - Deploy independently                                                                                                                                                                                    
                                                                                                                                                                                                            
  Communication                                                                                                                                                                                             
                                                                                                                                                                                                            
  - Synchronous: REST, gRPC for immediate responses                                                                                                                                                         
  - Asynchronous: Message queues for resilience                                                                                                                                                             
  - Events: Publish-subscribe for loose coupling                                                                                                                                                            
                                                                                                                                                                                                            
  Data Management                                                                                                                                                                                           
                                                                                                                                                                                                            
  - Each service owns its data                                                                                                                                                                              
  - Eventual consistency between services                                                                                                                                                                   
  - Saga patterns for distributed transactions                                                                                                                                                              
                                                                                                                                                                                                            
  Example Communication                                                                                                                                                                                     
                                                                                                                                                                                                            
  // Service A: User service                                                                                                                                                                                
  app.post('/users', async (req, res) => {                                                                                                                                                                  
    const user = await userService.createUser(req.body);                                                                                                                                                    
    await eventBus.publish('user.created', {                                                                                                                                                                
      userId: user.id,                                                                                                                                                                                      
      email: user.email                                                                                                                                                                                     
    });                                                                                                                                                                                                     
    res.json(user);                                                                                                                                                                                         
  });                                                                                                                                                                                                       
                                                                                                                                                                                                            
  // Service B: Email service consuming events                                                                                                                                                              
  eventBus.subscribe('user.created', async (data) => {                                                                                                                                                      
    await emailService.sendWelcome(data.email);                                                                                                                                                             
  });                                                                                                                                                                                                       
                                                                                                                                                                                                            
  Monolith-First Approach                                                                                                                                                                                   
                                                                                                                                                                                                            
  Start with monolithic architecture initially:                                                                                                                                                             
                                                                                                                                                                                                            
  Advantages                                                                                                                                                                                                
                                                                                                                                                                                                            
  - Easier to develop and deploy initially                                                                                                                                                                  
  - Simple horizontal scaling                                                                                                                                                                               
  - No distributed system complexity                                                                                                                                                                        
  - Faster iteration on business logic                                                                                                                                                                      
                                                                                                                                                                                                            
  When to Split                                                                                                                                                                                             
                                                                                                                                                                                                            
  - Teams becoming too large                                                                                                                                                                                
  - Deployment conflicts                                                                                                                                                                                    
  - Scaling specific components differently                                                                                                                                                                 
  - Technology diversity needs                                                                                                                                                                              
                                                                                                                                                                                                            
  Modular Monolith                                                                                                                                                                                          
                                                                                                                                                                                                            
  Structure monolith with clear boundaries:                                                                                                                                                                 
                                                                                                                                                                                                            
  src/                                                                                                                                                                                                      
  ├── user/                                                                                                                                                                                                 
  ├── order/                                                                                                                                                                                                
  ├── payment/                                                                                                                                                                                              
  ├── shared/                                                                                                                                                                                               
  └── main.js                                                                                                                                                                                               
                                                                                                                                                                                                            
  Apply architectural patterns within the monolith first, then extract services later.                                                                                                                      
                                                                                                                                                                                                            
  Implementation Guidelines                                                                                                                                                                                 
                                                                                                                                                                                                            
  Technology Selection                                                                                                                                                                                      
                                                                                                                                                                                                            
  Choose tools based on:                                                                                                                                                                                    
  - Team expertise                                                                                                                                                                                          
  - Long-term maintainability                                                                                                                                                                               
  - Performance requirements                                                                                                                                                                                
  - Community support                                                                                                                                                                                       
  - Security track record

  Handle errors consistently across layers:                                                                                                                                                                 
                                                                                                                                                                                                            
  // domain/errors/ValidationError.js                                                                                                                                                                       
  class ValidationError extends Error {                                                                                                                                                                     
    constructor(field, message) {                                                                                                                                                                           
      super(message);                                                                                                                                                                                       
      this.field = field;                                                                                                                                                                                   
      this.type = 'validation';                                                                                                                                                                             
    }                                                                                                                                                                                                       
  }                                                                                                                                                                                                         
                                                                                                                                                                                                            
  // application/middlewares/errorHandler.js                                                                                                                                                                
  const errorHandler = (err, req, res, next) => {                                                                                                                                                           
    if (err.type === 'validation') {

        error: 'Validation failed',                                                                                                                                                                         
        field: err.field,                                                                                                                                                                                   
        message: err.message                                                                                                                                                                                
      });                                                                                                                                                                                                   
    }                                                                                                                                                                                                       
                                                                                                                                                                                                            
    logger.error('Unexpected error', err);                                                                                                                                                                  
    res.status(500).json({ error: 'Internal server error' });                                                                                                                                               
  };

      .isLength({ min: 2, max: 50 })

      .matches(/^[a-zA-Z\s'-]+$/)                                                                                                                                                                           
      .withMessage('Name can only contain letters, spaces, hyphens, and apostrophes'),                                                                                                                      
                                                                                                                                                                                                            
    body('email')                                                                                                                                                                                           
      .isEmail()

      .isLength({ min: 12 })

      .matches(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]/)                                                                                                                           
      .withMessage('Password must contain uppercase, lowercase, number, and special character'),

          success: false,                                                                                                                                                                                   
          message: 'Validation failed',                                                                                                                                                                     
          errors: errors.array().map(error => ({                                                                                                                                                            
            field: error.path,                                                                                                                                                                              
            message: error.msg                                                                                                                                                                              
          }))                                                                                                                                                                                               
        });                                                                                                                                                                                                 
      }                                                                                                                                                                                                     
      next();                                                                                                                                                                                               
    }
